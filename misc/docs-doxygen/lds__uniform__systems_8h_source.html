144 template <typename System>
145 inline void UniformSystemList<System>::Swap(System& that, size_t n) {
146  // make sure request in range
147  if (n >= this->size()) {
148  std::cerr
149  << "Requested UniformSystemList element out of bounds. Skipping.\n";
150  return;
151  }
152  // check dim
153  bool does_match = (dim_[0] == that.n_u()) && (dim_[1] == that.n_x()) &&
154  (dim_[2] == that.n_y());
155  if (!does_match) {
156  std::cerr << "Cannot swap a UniformSystemList element for an element of "
157  "different size. Skipping.\n";
158  return;
159  }
160  // if checks pass, perform swap
161  System tmp = std::move((*this)[n]);
162  (*this)[n] = std::move(that);
163  that = std::move(tmp);
164 }

166 template <typename System>
167 inline UniformSystemList<System>& UniformSystemList<System>::operator=(
168  const UniformSystemList& that) {
169  // check dimensions
170  if (!this->empty()) {
171  if (this->size() != that.size()) {
172  std::ostringstream ss;
173  ss << "cannot reassign " << this->size() << " systems with "
174  << that.size() << " systems";
175  throw std::runtime_error(ss.str());
176  }
177 
178  if (dim_[0] + dim_[1] + dim_[2]) {
179  std::array<size_t, 3> other_dim(that.dim());
180  if (dim_ != other_dim) {
181  std::ostringstream ss;
182  ss << "cannot reassign systems of size " << dim_[0] << "x" << dim_[1]
183  << "x" << dim_[2] << " with systems of size " << other_dim[0] << "x"
184  << other_dim[1] << "x" << dim_[2];
185  throw std::runtime_error(ss.str());
186  }
187  }
188  }
189 
190  for (size_t k = 0; k < this->size(); k++) {
191  (*this)[k] = that[k];
192  }
193  return (*this);
194 }

197 inline UniformSystemList<System>& UniformSystemList<System>::operator=(
198  UniformSystemList&& that) noexcept {
199  // check dimensions
200  // if empty, assume a default constructed object and safe to move
201  if (!this->empty()) {
202  if (this->size() != that.size()) {
203  std::cerr << "Cannot reassign " << this->size() << " systems with "

263 template <typename System>
264 void UniformSystemList<System>::CheckDimensions(std::array<size_t, 3> dim) {
265  if (dim[0] + dim[1] + dim[2]) {
266  dim_ = dim;
267  } else {
268  dim_[0] = this->at(0).n_u();
269  dim_[1] = this->at(0).n_x();
270  dim_[2] = this->at(0).n_y();
271  }
272 
273  // make sure dimensiolaties are all uniform
274  bool does_match(true);
275  for (const System& sys : *this) {
276  does_match = does_match && (sys.n_u() == dim_[0]);
277  does_match = does_match && (sys.n_x() == dim_[1]);
278  does_match = does_match && (sys.n_y() == dim_[2]);
279  if (!does_match) {
280  throw std::runtime_error(
281  "Dimensionality of one or more input systems are not uniform.");
282  }
