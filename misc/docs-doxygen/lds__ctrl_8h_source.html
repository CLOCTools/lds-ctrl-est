361 template <typename System>
362 inline void Controller<System>::set_control_type(size_t control_type) {
363  if (control_type_ == control_type) {
364  return;
365  }
366 
367  // creating a blank slate...
368  control_type_ = 0;
369  Kc_inty_.zeros(0, 0);
370  Kc_u_.zeros(0, 0);
371  int_e_.zeros(0, 0);
372  int_e_awu_adjust_.zeros(0, 0);
373 
374  // controller was designed to minimize integral error
375  if (control_type & kControlTypeIntY) {
376  Kc_inty_.zeros(sys_.n_u(), sys_.n_y());
377  int_e_.zeros(sys_.n_y());
378  int_e_awu_adjust_.zeros(sys_.n_u());
379  control_type_ = control_type_ | kControlTypeIntY;
380  }
381 
382  // controller was designed to minimize deltaU
383  // (i.e. state augmented with u)
384  if (control_type & kControlTypeDeltaU) {
385  Kc_u_.zeros(sys_.n_u(), sys_.n_u());
386  control_type_ = control_type_ | kControlTypeDeltaU;
387  }
388 
389  // whether to adapt set point calculate with (re-estimated) process
390  // disturbance (m)
391  if (control_type & kControlTypeAdaptM) {
392  if (sys_.do_adapt_m) // only if adapting m...
393  {
394  control_type_ = control_type_ | kControlTypeAdaptM;
395  }
396  }

441 template <typename System>
442 inline void Controller<System>::CalcControl(bool do_control, bool do_estimation,
443  bool do_lock_control,
444  data_t sigma_soft_start,
445  data_t sigma_u_noise,
446  bool do_reset_at_control_onset) {
447  if (do_control && do_estimation) {
448  if (!do_control_prev_) {
449  if (do_reset_at_control_onset) {
450  Reset();
451  }
452  t_since_control_onset_ = 0.0;
453  } else {
454  t_since_control_onset_ += sys_.dt();
455  }
456 
457  // enforce softstart on control vars.
458  if (sigma_soft_start > 0) {
459  // half-Gaussian soft-start scaling factor
460  data_t soft_start_sf = 1 - exp(-pow(t_since_control_onset_, 2) /
461  (2 * pow(sigma_soft_start, 2)));
462  u_ref_ *= soft_start_sf;
463  // TODO(mfbolus): May be appropriate to soft-start x_ref, y_ref too
464  // x_ref_ *= soft_start_sf;
465  // cx_ref_ *= soft_start_sf;
466  // y_ref_ *= soft_start_sf;
467  }
468 
469  if (!do_lock_control) {
470  du_ref_ = u_ref_ - u_ref_prev_;
471 
472  // first do u -> v change of vars. (v = g.*u)
473  // e.g., convert into physical units (e.g., v[=] mW/mm2 rather than driver
474  // control voltage u[=]V)
475  v_ref_ = g_design_ % u_ref_;
476  dv_ref_ = g_design_ % du_ref_;
477 
478  // Given FB, calc. the change in control
479  if (control_type_ & kControlTypeDeltaU) {
480  // if control designed to minimize not u but deltaU (i.e. state aug with
481  // u):
482  dv_ = dv_ref_; // nominally-optimal.
483  dv_ -= Kc_ * (sys_.x() - x_ref_); // instantaneous state error
484  dv_ -= Kc_u_ * (v_ - v_ref_); // penalty on amp u (rel to ref)
485 
486  if (control_type_ & kControlTypeIntY) {
487  // TODO(mfbolus): one approach to protection against integral windup
488  // would be to not integrate error when control signal saturated:
489 
490  // if(!uSaturated)
491  int_e_ += (sys_.cx() - cx_ref_) * sys_.dt(); // integrated error
492  dv_ -= Kc_inty_ * int_e_; // control for integrated error
493  }
494 
495  // update the control
496  v_ += dv_;
497  } else {
498  v_ = v_ref_; // nominally-optimal.
499  v_ -= Kc_ * (sys_.x() - x_ref_); // instantaneous state error
500 
501  if (control_type_ & kControlTypeIntY) {
502  // TODO(mfbolus): one approach to protection against integral windup
503  // would be to not integrate error when control signal saturated:
504 
505  // if (!uSaturated)
506  int_e_ += (sys_.cx() - cx_ref_) * sys_.dt(); // integrated error
507  v_ -= Kc_inty_ * int_e_; // control for integrated error
508  }
509  }
510 
511  // convert back to control voltage u[=]V
512  u_ = v_ / sys_.g();
513  } // else do nothing until lock is low
514 
515  // It may be desireable to make inputs more variable.
516  if (sigma_u_noise > 0.0) {
517  u_ += sigma_u_noise * Vector(sys_.n_u(), fill::randn);
518  };
519 
520  } else { // if not control
521  // feed through u_ref in open loop
522  u_ = u_ref_ % g_design_ / sys_.g();
523  u_ref_.zeros();
524  int_e_.zeros();
525  int_e_awu_adjust_.zeros();
526  u_sat_.zeros();
527  } // ends do_control
528 
529  // enforce box constraints (and antiwindup)
530  AntiWindup();
531 
532  do_control_prev_ = do_control;
533  do_lock_control_prev_ = do_lock_control;
534 } // CalcControl

536 template <typename System>
537 inline void Controller<System>::CalcSteadyStateSetPoint() {
538  // Linearly-constrained least squares (ls).
539  //
540  // _reference:
541  // Boyd & Vandenberghe (2018) Introduction to Applied Linear Algebra
542  //
543  Matrix a_ls =
544  join_horiz(sys_.C(), Matrix(sys_.n_y(), sys_.n_u(), fill::zeros));
545  Vector b_ls = cx_ref_;
546  Matrix c_ls = join_horiz(sys_.A() - Matrix(sys_.n_x(), sys_.n_x(), fill::eye),
547  sys_.B() * arma::diagmat(sys_.g()));
548  Vector d_ls = -sys_.m0();
549  if (control_type_ & kControlTypeAdaptM) {
550  d_ls = -sys_.m(); // adapt setpoint calc with disturbance?
551  }
552 
553  Matrix a_ls_t = a_ls.t(); // TODO(mfbolus): not sure why but causes seg
554  // fault if I do not do this.
555  Matrix phi_ls =
556  join_vert(join_horiz(2 * a_ls_t * a_ls, c_ls.t()),
557  join_horiz(c_ls, Matrix(sys_.n_x(), sys_.n_x(), fill::zeros)));
558  // TODO(mfbolus): should be actual inverse, rather than pseudo-inverse:
559  Matrix inv_phi = pinv(phi_ls);
560  Vector xulam = inv_phi * join_vert(2 * a_ls_t * b_ls, d_ls);
561  x_ref_ = xulam.subvec(0, sys_.n_x() - 1);
562  u_ref_ = xulam.subvec(sys_.n_x(), sys_.n_x() + sys_.n_u() - 1);
563  cx_ref_ = sys_.C() * x_ref_;
564 } // CalcSteadyStateSetPoint

566 template <typename System>
567 void Controller<System>::AntiWindup() {
568  u_saturated_ = false;
569  u_sat_ = u_;
570 
571  // limit u and flag whether saturated
572  for (size_t k = 0; k < u_.n_elem; k++) {
573  if (u_[k] < u_lb_) {
574  u_sat_[k] = u_lb_;
575  u_saturated_ = true;
576  }
577 
578  if (u_[k] > u_ub_) {
579  u_sat_[k] = u_ub_;
580  u_saturated_ = true;
581  }
582  }
583 
584  if ((control_type_ & kControlTypeIntY) && (tau_awu_ < lds::kInf)) {
585  // one-step back-calculation (calculate intE for u=u_sat)
586  // (Astroem, Rundqwist 1989 warn against using this...)
587  // int_e_awu_adjust_ =
588  // solve(Kc_inty_, (u_ - u_sat_)); // pinv(Kc_inty) * (u-uSat);
589 
590  // gradual: see Astroem, Rundqwist 1989
591  // this is a fudge for doing MIMO gradual
592  // n.b., went ahead and multiplied 1/T by dt so don't have to do that here.
593  int_e_awu_adjust_ =
594  k_awu_ * (sign(Kc_inty_).t() / sys_.n_u()) * (u_ - u_sat_);
595  // int_e_awu_adjust_ = k_awu_ * (u_-u_sat_);
596 
597  int_e_ += int_e_awu_adjust_;
598  }
599 
600  // set u to saturated version
601  u_ = u_sat_;
602 }
