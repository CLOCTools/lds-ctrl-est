50 auto m2a_cellmat(matlab::data::CellArray& matlab_cell)
51  -> std::vector<arma::Mat<T>> {
52  size_t n_cells = matlab_cell.getNumberOfElements();
53  std::vector<arma::Mat<T>> arma_mat(n_cells,
54  arma::mat(1, 1, arma::fill::zeros));
55  for (size_t k = 0; k < n_cells; k++) {
56  matlab::data::TypedArray<T> matlab_mat = matlab_cell[k];

59  }
60  return arma_mat;
