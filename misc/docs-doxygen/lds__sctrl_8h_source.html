232 template <typename System>
233 inline void SwitchedController<System>::Switch(size_t idx,
234  bool do_force_switch) {
235  if ((idx == idx_) && !do_force_switch) {
236  return; // already there.
237  }
238 
239  // put old up and get new one out
240  systems_.at(idx_) = std::move(sys_);
241  sys_ = std::move(systems_.at(idx));
242 
243  Kc_list_.Swap(Kc_, idx_);
244  Kc_list_.Swap(Kc_, idx);
245 
246  if (control_type_ & kControlTypeIntY) {
247  Kc_inty_list_.Swap(Kc_inty_, idx_);
248  Kc_inty_list_.Swap(Kc_inty_, idx);
249  }
250 
251  if (control_type_ & kControlTypeDeltaU) {
252  Kc_u_list_.Swap(Kc_u_, idx_);
253  Kc_u_list_.Swap(Kc_u_, idx);
254  }
255 
256  g_design_list_.Swap(g_design_, idx_);
257  g_design_list_.Swap(g_design_, idx);
258 
259  idx_ = idx;
260 } // Switch
