151 template <typename Fit>
152 SSID<Fit>::SSID(size_t n_x, size_t n_h, data_t dt,
153  UniformMatrixList<kMatFreeDim2>&& u_train,
154  UniformMatrixList<kMatFreeDim2>&& z_train, const Vector& d) {
155  // check input/output data dimensions are consistent
156  if (z_train.size() != u_train.size()) {
157  throw std::runtime_error(
158  "I/O training data have different number of trials.");
159  }
160  n_trials_ = u_train.size();
161 
162  n_t_tot_ = 0;
163  n_t_ = std::vector<size_t>(n_trials_);
164  for (size_t trial = 0; trial < n_trials_; trial++) {
165  if (z_train.at(trial).n_cols != u_train.at(trial).n_cols) {
166  throw std::runtime_error(
167  "I/O training data have different number of time steps.");
168  }
169  n_t_[trial] = u_train.at(trial).n_cols;
170  n_t_tot_ += n_t_[trial];
171  }
172 
173  dt_ = dt;
174  n_x_ = n_x;
175  n_u_ = u_train.at(0).n_rows;
176  n_y_ = z_train.at(0).n_rows;
177  n_h_ = n_h;
178 
179  // dimensionality check for eventual block-hankel data matrix
180  size_t len = n_t_tot_ - 2 * n_h_ + 1;
181  if (len < (2 * n_h_ * (n_u_ + n_y_))) {
182  std::ostringstream ss;
183  ss << "Dataset problem! More rows than columns in block-hankel data "
184  "matrix: 2*(n_u+n_y)*n_h > data-length! Need higher data-length or "
185  "lower n_h.";
186  throw std::runtime_error(ss.str());
187  }
188 
189  fit_ = Fit(n_u_, n_x_, n_y_, dt_);
190 
191  u_ = std::move(u_train);
192  z_ = std::move(z_train);
193 
194  if (!d.is_finite() || (d.n_rows != n_y_)) {
195  // TODO(mfbolus): implement least-square solution for impulse response with
196  // a second input of ones. Data-driven way of accounting for offset *not*
197  // driven by an input.
198  //
199  // For now, calculate output bias (d) as the
200  // output wherever the stimulus has not been on for some amount of time.
201  // convolve u with rectangle and take all samples. This is a reasonable
202  // approach, since often when autonomous systems are fit (i.e., systems with
203  // no input), they will subtract off the mean of the output. This
204  // essentially amounts to setting output bias to the mean of the output when
205  // there is no stimulation.
206  data_t t_silence = 0.1;
207  data_t thresh_silence = 0.001;
208  CalcD(t_silence, thresh_silence);
209  } else {
210  fit_.set_d(d);
211  }
212 }

215 std::tuple<Fit, Vector> SSID<Fit>::Run(SSIDWt ssid_wt) {
216  // the weight on minimizing dc I/O gain only works for gaussian,
217  // and hopefully not necessary with appropriate dataset.
218  data_t wt_dc = 0;
219  CreateHankelDataMat();

226 template <typename Fit>
227 void SSID<Fit>::CalcD(data_t t_silence, data_t thresh_silence) {
228  Vector d(z_.at(0).n_rows, fill::zeros);
229  Vector win(static_cast<size_t>(t_silence / dt_), fill::ones);
230  Vector sum_z_silence(n_y_, fill::zeros);
231  size_t n_silence(0);
232  for (size_t trial = 0; trial < u_.size(); trial++) {
233  // find silent samples
234  // start by convolving with
235  Vector sum_u = vectorise(sum(abs(u_.at(trial)), 0));
236  Vector u_conv = conv(sum_u, win, "same");
237 
238  // get only the samples that are silent...
239  arma::uvec ubi_silence = find(u_conv <= thresh_silence);
240  if (ubi_silence.n_elem > 0) {
241  sum_z_silence += arma::sum(z_.at(trial).cols(ubi_silence), 1);
242  n_silence += ubi_silence.n_elem;
243  }
244  }
245  if (n_silence > 0) {
246  d = sum_z_silence / n_silence;
247  }
248  fit_.set_d(d);
249 }

251 template <typename Fit>
252 void SSID<Fit>::CreateHankelDataMat() {
253  // temporary copy of data
254  Matrix z(n_y_, n_t_tot_, fill::zeros);
255  Matrix u(n_u_, n_t_tot_, fill::zeros);
256  size_t so_far(0);
257  for (size_t trial = 0; trial < z_.size(); trial++) {
258  z.submat(0, so_far, n_y_ - 1, so_far + n_t_.at(trial) - 1) = z_.at(trial);
259  u.submat(0, so_far, n_u_ - 1, so_far + n_t_.at(trial) - 1) = u_.at(trial);
260  so_far += n_t_.at(trial);
261  }
262  // remove output bias
263  z.each_col() -= fit_.d();
264 
265  // calculate I/O gain @ DC while data in convenient form
266  g_dc_ = z * pinv(u);
267 
268  // create hankel data matrix
269  size_t len = z.n_cols - 2 * n_h_ + 1; // data length in hankel mat
270 
271  // block-hankel data matrix
272  D_ = Matrix(2 * n_h_ * (n_u_ + n_y_), len);
273  // past input
274  auto u_p = D_.submat(0, 0, n_h_ * n_u_ - 1, len - 1);
275  // future input
276  auto u_f = D_.submat(n_h_ * n_u_, 0, 2 * n_h_ * n_u_ - 1, len - 1);
277  // past output
278  auto y_p =
279  D_.submat(2 * n_h_ * n_u_, 0, n_h_ * (2 * n_u_ + n_y_) - 1, len - 1);
280  // future output
281  auto y_f = D_.submat(n_h_ * (2 * n_u_ + n_y_), 0,
282  2 * n_h_ * (n_u_ + n_y_) - 1, len - 1);
283 
284  size_t idx = 0;
285  for (size_t k = 0; k < len; k++) {
286  idx = 0;
287  for (size_t kk = k; kk < (n_h_ + k); kk++) {
288  u_p.col(k).subvec(idx, idx + n_u_ - 1) = u.col(kk);
289  idx += n_u_;
290  }
291 
292  idx = 0;
293  for (size_t kk = (n_h_ + k); kk < (2 * n_h_ + k); kk++) {
294  u_f.col(k).subvec(idx, idx + n_u_ - 1) = u.col(kk);
295  idx += n_u_;
296  }
297 
298  idx = 0;
299  for (size_t kk = k; kk < (n_h_ + k); kk++) {
300  y_p.col(k).subvec(idx, idx + n_y_ - 1) = z.col(kk);
301  idx += n_y_;
302  }
303 
304  idx = 0;
305  for (size_t kk = (n_h_ + k); kk < (2 * n_h_ + k); kk++) {
306  y_f.col(k).subvec(idx, idx + n_y_ - 1) = z.col(kk);
307  idx += n_y_;
308  }
309  }
310 
311  // TODO(mfbolus): subtract mean?
312  // Vector m = mean(D_,1);
313  // D_.each_col() -= m;
314 
315  D_ /= sqrt(static_cast<data_t>(len));
316 }

318 template <typename Fit>
319 void SSID<Fit>::DecomposeData() {
320  // // do LQ decomp instead of calculating covariance expensive way
321  // // Note that "R" in van Overschee is lower-triangular (L), not "R" in QR
322  // // decomp. Very confusing.
323  // Matrix q_t;
324  // lq(L_, q_t, D_);
325 
326  // // van Overschee zeros out the other elements.
327  // L_ = trimatl(L_);
328 
329  // Depending on dataset, this may be faster:
330  // Calculate covariance of the data matrix
331  Matrix cov = D_ * D_.t();
332  // retro-actively do mean subtraction
333  Vector mu_d = arma::mean(D_, 1);
334  cov -= mu_d * mu_d.t() * D_.n_cols;
335  L_ = arma::chol(cov, "lower");
336  L_ = trimatl(L_);
337 }

339 template <typename Fit>
340 void SSID<Fit>::CalcSVD(SSIDWt wt) {
341  // submats that will be needed:
342  auto R_14_14 = L_.submat(0, 0, n_h_ * (2 * n_u_ + n_y_) - 1,
343  n_h_ * (2 * n_u_ + n_y_) - 1);
344  auto R_11_14 = L_.submat(0, 0, n_h_ * n_u_ - 1, n_h_ * (2 * n_u_ + n_y_) - 1);
345  auto R_11_13 = L_.submat(0, 0, n_h_ * n_u_ - 1, n_h_ * (2 * n_u_) - 1);
346  auto R_23_13 =
347  L_.submat(n_h_ * n_u_, 0, 2 * n_h_ * n_u_ - 1, 2 * n_h_ * n_u_ - 1);
348  auto R_44_14 = L_.submat(2 * n_u_ * n_h_, 0, n_h_ * (2 * n_u_ + n_y_) - 1,
349  n_h_ * (2 * n_u_ + n_y_) - 1);
350  auto R_44_13 = L_.submat(2 * n_u_ * n_h_, 0, n_h_ * (2 * n_u_ + n_y_) - 1,
351  n_h_ * (2 * n_u_) - 1);
352  auto R_44 =
353  L_.submat(2 * n_u_ * n_h_, 2 * n_u_ * n_h_, n_h_ * (2 * n_u_ + n_y_) - 1,
354  n_h_ * (2 * n_u_ + n_y_) - 1);
355  auto R_56_14 =
356  L_.submat(n_h_ * (2 * n_u_ + n_y_), 0, n_h_ * (2 * n_u_ + 2 * n_y_) - 1,
357  n_h_ * (2 * n_u_ + n_y_) - 1);
358 
359  Matrix Lup_Luf_Lyp = R_56_14 * pinv(R_14_14);
360  auto Lup = Lup_Luf_Lyp.submat(0, 0, n_h_ * n_y_ - 1, n_h_ * n_u_ - 1);
361  auto Luf =
362  Lup_Luf_Lyp.submat(0, n_h_ * n_u_, n_h_ * n_y_ - 1, 2 * n_h_ * n_u_ - 1);
363  auto Lyp = Lup_Luf_Lyp.submat(0, 2 * n_h_ * n_u_, n_h_ * n_y_ - 1,
364  n_h_ * (2 * n_u_ + n_y_) - 1);
365 
366  // aka: R_f
367  Matrix R_56_16 = L_.submat(n_h_ * (2 * n_u_ + n_y_), 0,
368  2 * n_h_ * (n_u_ + n_y_) - 1, L_.n_cols - 1);
369  // from van Overschee subid.m:
370  // Rf = R((2*m+l)*i+1:2*(m+l)*i,:); % Future outputs
371 
372  Matrix U;
373  Matrix V;
374  switch (wt) {
375  case kSSIDNone: {
376  // No weighting. (what van Overschee calls "N4SID")
377  Matrix O_k_sans_Qt = Lup * R_11_14 + Lyp * R_44_14;
378  arma::svd(U, s_, V, O_k_sans_Qt);
379  } break;
380  case kSSIDMOESP: {
381  // MOESP weighting
382  // This is what they use in the "robust" algorithm van Overschee, de Moor
383  // 1996
384  Matrix Pi = Matrix(2 * n_h_ * n_u_, 2 * n_h_ * n_u_, fill::eye) -
385  R_23_13.t() * inv(R_23_13 * R_23_13.t()) * R_23_13;
386  Matrix O_k_ortho_Uf_sans_Qt =
387  join_horiz((Lup * R_11_13 + Lyp * R_44_13) * Pi, Lyp * R_44);
388  svd(U, s_, V, O_k_ortho_Uf_sans_Qt);
389  } break;
390  case kSSIDCVA: {
391  // CVA weighting
392  // See van Overschee's matlab code (subid.m):
393  // https://www.mathworks.com/matlabcentral/fileexchange/2290-subspace-identification-for-linear-systems
394  Matrix Pi = Matrix(2 * n_h_ * n_u_, 2 * n_h_ * n_u_, fill::eye) -
395  R_23_13.t() * inv(R_23_13 * R_23_13.t()) * R_23_13;
396  Matrix O_k_ortho_Uf_sans_Qt =
397  join_horiz((Lup * R_11_13 + Lyp * R_44_13) * Pi, Lyp * R_44);
398 
399  Matrix inv_w1;
400  Matrix qt1;
401  lq(inv_w1, qt1, R_56_16); // lq decomp of R_f (future output data)
402  inv_w1 = trimatl(inv_w1);
403  inv_w1 = inv_w1.submat(0, 0, n_y_ * n_h_ - 1, n_y_ * n_h_ - 1);
404  Matrix w_o_w = arma::solve(
405  inv_w1, O_k_ortho_Uf_sans_Qt); // alternatively
406  // pinv(inv_W1)*O_k_ortho_Uf_sans_Qt
407  svd(U, s_, V, w_o_w);
408 
409  U = inv_w1 * U;
410  break;
411  }
412  }
413 
414  // Truncate to model order (heart of ssid method)
415  auto s_hat = s_.subvec(0, n_x_ - 1);
416  Matrix diag_sqrt_s = diagmat(sqrt(s_hat));
417  auto u_hat = U.submat(0, 0, U.n_rows - 1, n_x_ - 1);
418 
419  // get extended observability and controllability mats
420  ext_obs_t_ = u_hat * diag_sqrt_s; // extended observability matrix
421 }

423 template <typename Fit>
424 void SSID<Fit>::Solve(data_t wt_dc) {
425  // required submats
426  auto R_56_14 =
427  L_.submat(n_h_ * (2 * n_u_ + n_y_), 0, n_h_ * (2 * n_u_ + 2 * n_y_) - 1,
428  n_h_ * (2 * n_u_ + n_y_) - 1);
429  auto R_23_15 = L_.submat(n_h_ * n_u_, 0, 2 * n_h_ * n_u_ - 1,
430  n_h_ * (2 * n_u_ + n_y_) + n_y_ - 1);
431  auto R_66_15 = L_.submat(n_h_ * (2 * n_u_ + n_y_) + n_y_, 0,
432  2 * n_h_ * (n_u_ + n_y_) - 1,
433  n_h_ * (2 * n_u_ + n_y_) + n_y_ - 1);
434  auto R_55_15 = L_.submat(n_h_ * (2 * n_u_ + n_y_), 0,
435  n_h_ * (2 * n_u_ + n_y_) + n_y_ - 1,
436  n_h_ * (2 * n_u_ + n_y_) + n_y_ - 1);
437  auto R_56_15 =
438  L_.submat(n_h_ * (2 * n_u_ + n_y_), 0, 2 * n_h_ * (n_u_ + n_y_) - 1,
439  n_h_ * (2 * n_u_ + n_y_) + n_y_ - 1);
440 
441  // Solve for params using appropriate algorithm:
442  // robust deterministic/stochastic algorithm in van Overschee 1996
443  // algorithm that the authors say "works" in practice.
444  auto ext_obs_tm1 = ext_obs_t_.submat(
445  0, 0, ext_obs_t_.n_rows - 1 - n_y_,
446  ext_obs_t_.n_cols - 1); // extended observability matrix
447 
448  // This is what textbook (1996) says:
449  //
450  // Matrix Tr = join_vert(pinv(ext_obs_t_) * R_56_15, R_23_15);
451  //
452  // HOWEVER, do not know why but have to fill the last place with zeros like
453  // authors' matlab implementation (see `subid.m`)
454  // Otherwise, get ridiculous covariances (although A,C estimates are close to
455  // same...)
456  Matrix Tr = join_vert(
457  join_horiz(pinv(ext_obs_t_) * R_56_14, Matrix(n_x_, n_y_, fill::zeros)),
458  R_23_15);
459  Matrix Tl = join_vert(pinv(ext_obs_tm1) * R_66_15, R_55_15);
460  Matrix S = Tl * pinv(Tr);
461 
462  // Use alternative in van Overschee 1996, p. 129. Apparently, should ensure
463  // stability.
464  fit_.set_C(ext_obs_t_.submat(0, 0, n_y_ - 1, ext_obs_t_.n_cols - 1));
465  Matrix ext_obs_t_p1 = join_vert(
466  ext_obs_t_.submat(n_y_, 0, ext_obs_t_.n_rows - 1, ext_obs_t_.n_cols - 1),
467  Matrix(n_y_, ext_obs_t_.n_cols, fill::zeros));
468  fit_.set_A(pinv(ext_obs_t_) * ext_obs_t_p1);
469 
470  // At this point, van Overschee & de Moor suggest re-calculating ext_obs_t_,
471  // ext_obs_tm1 from (A, C) because it was just an approximation. This is
472  RecomputeExtObs();
473  ext_obs_tm1 = ext_obs_t_.submat(
474  0, 0, ext_obs_t_.n_rows - 1 - n_y_,
475  ext_obs_t_.n_cols - 1); // extended observability matrix
476  Tl = join_vert(pinv(ext_obs_tm1) * R_66_15, R_55_15);
477  Tr = join_vert(
478  join_horiz(pinv(ext_obs_t_) * R_56_14, Matrix(n_x_, n_y_, fill::zeros)),
479  R_23_15);
480  S = Tl * pinv(Tr);
481 
482  Matrix Lcurly = S.submat(0, 0, n_x_ + n_y_ - 1, n_x_ - 1) * pinv(ext_obs_t_);
483  Matrix Mcurly = pinv(ext_obs_tm1);
484  Matrix Pcurly = Tl - Lcurly * R_56_15;
485  Vector Pvec = vectorise(Pcurly);
486  Matrix Qcurly = R_23_15;
487 
488  // Identify [D; B], assuming D=0 and ensuring DC gain is correct
489  Matrix sum_QcurlyT_kron_Ncurly(
490  (n_h_ * (2 * n_u_ + n_y_) + n_y_) * (n_y_ + n_x_), n_u_ * (n_y_ + n_x_),
491  fill::zeros);
492 
493  Matrix eye_ext_obs_tm1(n_y_ + ext_obs_tm1.n_rows, n_y_ + ext_obs_tm1.n_cols,
494  fill::eye);
495  eye_ext_obs_tm1.submat(n_y_, n_y_, eye_ext_obs_tm1.n_rows - 1,
496  eye_ext_obs_tm1.n_cols - 1) = ext_obs_tm1;
497 
498  // van Overschee (1996) p. 126
499  Matrix N1_Tl = -Lcurly;
500  N1_Tl.submat(0, 0, n_x_ - 1, N1_Tl.n_cols - 1) +=
501  join_horiz(Matrix(n_x_, n_y_, fill::zeros), Mcurly);
502  N1_Tl.submat(n_x_, 0, n_x_ + n_y_ - 1, n_y_ - 1) +=
503  Matrix(n_y_, n_y_, fill::eye);
504 
505  Matrix Nk_Tl(N1_Tl.n_rows, N1_Tl.n_cols, fill::zeros);
506  Matrix N_k;
507  for (size_t k = 0; k < n_h_; k++) {
508  auto Qcurly_k =
509  Qcurly.submat(n_u_ * k, 0, n_u_ * (k + 1) - 1, Qcurly.n_cols - 1);
510 
511  Nk_Tl.zeros();
512  Nk_Tl.submat(0, 0, n_x_ + n_y_ - 1, Nk_Tl.n_cols - k * n_y_ - 1) =
513  N1_Tl.submat(0, k * n_y_, N1_Tl.n_rows - 1, N1_Tl.n_cols - 1);
514  N_k = Nk_Tl * eye_ext_obs_tm1;
515 
516  sum_QcurlyT_kron_Ncurly += kron(Qcurly_k.t(), N_k);
517  }
518 
519  Matrix err_vec;
520  if (wt_dc > 0) {
521  // Constraints enforced by weighted least squares
522  //
523  // Reference:
524  //
525  // Privara S, ..., Ferkl L_. (2010) Subspace Identification of Poorly
526  // Excited Industrial Systems. Conference in Decision and Control.
527 
528  // constraint 1: assume D=0 --> remove the components for Dvec (this is
529  // actually a hard constraint in that it ignores D)
530  Matrix sum_QcurlyT_kron_Ncurly_db = sum_QcurlyT_kron_Ncurly;
531  sum_QcurlyT_kron_Ncurly =
532  Matrix(sum_QcurlyT_kron_Ncurly_db.n_rows, n_x_ * n_u_);
533 
534  size_t kkk = 0;
535  for (size_t k = 1; k < (n_u_ + 1); k++) {
536  size_t start_idx = k * (n_y_ + n_x_) - n_x_;
537  for (size_t kk = 0; kk < n_x_; kk++) {
538  sum_QcurlyT_kron_Ncurly.col(kkk) =
539  sum_QcurlyT_kron_Ncurly_db.col(start_idx + kk);
540  kkk++;
541  }
542  }
543 
544  // constraint 2: Make sure DC I/O gain is correct
545  Matrix b_to_g0 = fit_.C() * inv(Matrix(n_x_, n_x_, fill::eye) - fit_.A());
