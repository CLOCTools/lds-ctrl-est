290 const Fit& EM<Fit>::Run(bool calc_dynamics, bool calc_Q, bool calc_init,
291  bool calc_output, bool calc_measurement,
292  size_t max_iter, data_t tol) {
293  Reset(); // to initial conditions
294 
295  size_t n_params =
296  3 * n_x_ * n_x_ + n_x_ * n_u_ + n_x_ + n_y_ * n_x_ + n_y_ * n_y_;
297  Vector theta(n_params);
298  Vector theta_new(n_params);
299  data_t max_dtheta = 1;
300 
301  // if solving for initial conditions, allow them be varied.
302  // otherwise, freeze at provided values.
303  bool force_common_initial = !calc_init;
304 
305  // go until parameter convergence
306  for (size_t l = 0; l < max_iter; l++) {
307  theta_ = UpdateTheta();
308 
309  std::cout << "Iteration " << l + 1 << "/" << max_iter << " ...\n";
310 
311  Expectation(force_common_initial);
312  Maximization(calc_dynamics, calc_Q, calc_init, calc_output,
313  calc_measurement);
314 
315  // check convergence
316  theta_new = UpdateTheta();
317 
318  Vector dtheta = abs(theta_new - theta_) / abs(theta_);
319  // some parameters could be zero...
320  arma::uvec ubi_finite = find_finite(dtheta);
321 
322  max_dtheta = max(dtheta.elem(ubi_finite));
323  std::cout << "max dtheta: " << max_dtheta << "\n";
324  if (max_dtheta < tol) {
325  std::cout << "Converged.\n";
326  break;
327  }
328 
329  std::cout << "\n";
330  }
331 
332  return fit_;
333 }

335 template <typename Fit>
336 void EM<Fit>::Smooth(bool force_common_initial) {
337  Matrix k_e(n_x_, n_y_); // estimator gain
338  Cube k_backfilt; // back-filtering gains
339 
340  // TODO(mfbolus): this loop could be made parallel
341  for (size_t trial = 0; trial < z_.size(); trial++) {
342  Matrix x_pre(n_x_, n_t_[trial], fill::zeros);
343  Cube p_pre(n_x_, n_x_, n_t_[trial], fill::zeros);
344  Matrix x_post(n_x_, n_t_[trial], fill::zeros);
345  Cube p_post(n_x_, n_x_, n_t_[trial], fill::zeros);
346 
347  if (force_common_initial) // forces all trials to have same initial
348  // conditions.
349  {
350  x_[trial].col(0) = fit_.x0();
351  P_[trial].slice(0) = fit_.P0();
352  }
353  y_[trial].col(0) = fit_.C() * x_[trial].col(0) + fit_.d();
354 
355  // This *should not* be necessary but make sure P is symmetric.
356  ForceSymPD(P_[trial].slice(0));
357 
358  x_pre.col(0) = x_[trial].col(0);
359  p_pre.slice(0) = P_[trial].slice(0);
360 
361  x_post.col(0) = x_[trial].col(0);
362  p_post.slice(0) = P_[trial].slice(0);
363 
364  // filter
365  for (size_t t = 1; t < n_t_[trial]; t++) {
366  // predict
367  fit_.f(x_pre, x_post, u_.at(trial), t);
368  fit_.h(y_[trial], x_pre, t);
369  diag_y_.diag() = y_[trial].col(t); // TODO(mfbolus): change if parallel
370 
371  // update --> posterior estimation
372  RecurseKe(k_e, p_pre, p_post, t);
373  x_post.col(t) =
374  x_pre.col(t) + k_e * (z_.at(trial).col(t) - y_[trial].col(t));
375  y_[trial].col(t) = fit_.C() * x_post.col(t) + fit_.d();
376  }
377  ForceSymPD(P_[trial].slice(n_t_[trial] - 1));
378 
379  // backfilter -> Smoothed estimate
380  // Reference:
381  // Shumway et Stoffer (1982)
382  k_backfilt = Cube(n_x_, n_x_, n_t_[trial], fill::zeros);
383  x_[trial].col(n_t_[trial] - 1) = x_post.col(n_t_[trial] - 1);
384  P_[trial].slice(n_t_[trial] - 1) = p_post.slice(n_t_[trial] - 1);
385  for (size_t t = (n_t_[trial] - 1); t > 0; t--) {
386  k_backfilt.slice(t - 1) =
387  p_post.slice(t - 1) * fit_.A().t() * inv_sympd(p_pre.slice(t));
388  x_[trial].col(t - 1) =
389  x_post.col(t - 1) +
390  k_backfilt.slice(t - 1) * (x_[trial].col(t) - x_pre.col(t));
391  P_[trial].slice(t - 1) =
392  p_post.slice(t - 1) + k_backfilt.slice(t - 1) *
393  (P_[trial].slice(t) - p_pre.slice(t)) *
394  k_backfilt.slice(t - 1).t();
395  }
396 
397  // do the same for P_t_tm1
398  Matrix id(n_x_, n_x_, fill::eye);
399  P_t_tm1_[trial].slice(n_t_[trial] - 1) =
400  (id - k_e * fit_.C()) * fit_.A() * p_post.slice(n_t_[trial] - 2);
401  for (size_t t = (n_t_[trial] - 1); t > 1; t--) {
402  P_t_tm1_[trial].slice(t - 1) =
403  p_post.slice(t - 1) * k_backfilt.slice(t - 2).t() +
404  k_backfilt.slice(t - 1) *
405  (P_t_tm1_[trial].slice(t) - fit_.A() * p_post.slice(t - 1)) *
406  k_backfilt.slice(t - 2).t();
407  }
408 
409  // finally, get smoothed estimate of output
410  for (size_t t = 0; t < n_t_[trial]; t++) {
411  fit_.h(y_[trial], x_[trial], t);
412  } // samps loop
413  } // trial loop
414 } // Smooth

439 template <typename Fit>
440 void EM<Fit>::Expectation(bool force_common_initial) {
441  // calculate the mean/cov of state needed for maximizing E[pr(z|theta)]
442  Smooth(force_common_initial);
443 
444  // now get the various forms of sum(E[xx']) needed
445  // n.b. Going to start at t=1 rather than 0 bc most max terms need that.
446  // so really "n_t_tot_" is (n_t_tot_-1)
447  n_t_tot_ = 0;
448  sum_E_x_t_x_t_ = Matrix(n_x_, n_x_, fill::zeros);
449  sum_E_xu_tm1_xu_tm1_ = Matrix(n_x_ + n_u_, n_x_ + n_u_, fill::zeros);
450  sum_E_xu_t_xu_tm1_ = Matrix(n_x_ + n_u_, n_x_ + n_u_, fill::zeros);
451 
452  Vector xu_tm1(n_x_ + n_u_, fill::zeros);
453  Vector xu_t(n_x_ + n_u_, fill::zeros);
454 
455  for (size_t trial = 0; trial < z_.size(); trial++) {
456  for (size_t t = 1; t < n_t_[trial]; t++) {
457  // ------------ sum_E_x_t_x_t ------------
458  sum_E_x_t_x_t_ += x_[trial].col(t) * x_[trial].col(t).t();
459  sum_E_x_t_x_t_ += P_[trial].slice(t);
460 
461  // ------------ sum_E_xu_tm1_xu_tm1 ------------
462  xu_tm1 = join_vert(x_[trial].col(t - 1), u_.at(trial).col(t - 1));
463  sum_E_xu_tm1_xu_tm1_ += xu_tm1 * xu_tm1.t();
464  sum_E_xu_tm1_xu_tm1_.submat(0, 0, n_x_ - 1, n_x_ - 1) +=
465  P_[trial].slice(t - 1);
466 
467  // ------------ sum_E_xu_t_xu_tm1 ------------
468  xu_t = join_vert(x_[trial].col(t), u_.at(trial).col(t));
469  sum_E_xu_t_xu_tm1_ += xu_t * xu_tm1.t();
470  sum_E_xu_t_xu_tm1_.submat(0, 0, n_x_ - 1, n_x_ - 1) +=
471  P_t_tm1_[trial].slice(t);
472 
473  n_t_tot_ += 1;
474  } // time
475  } // trial
476 } // Expectation

513 template <typename Fit>
514 void EM<Fit>::MaximizeQ() {
515  // // Shumway, Stoffer (1982); Ghahgramani, Hinton (1996)
516  // View sum_e_x_t_xu_tm1 =
517  // sum_E_xu_t_xu_tm1_.submat(0, 0, n_x_ - 1, n_x_ + n_u_ - 1);
518  // Matrix q = sum_E_x_t_x_t_ - sum_e_x_t_xu_tm1 *
519  // inv_sympd(sum_E_xu_tm1_xu_tm1_) *
520  // sum_e_x_t_xu_tm1.t();
521  // q /= n_t_tot_;
522 
523  // this way is same as above iff dynamics were just updated:
524  // View sum_e_x_t_xu_tm1 =
525  // sum_E_xu_t_xu_tm1_.submat(0, 0, n_x_ - 1, n_x_ + n_u_ - 1);
526  // Matrix ab = arma::join_horiz(fit_.A(), fit_.B());
527  // Matrix q = sum_E_x_t_x_t_ - ab * sum_e_x_t_xu_tm1.t();
528  // q /= n_t_tot_;
529 
530  // From scratch method:
531  // Q is covariance of the error between state and dynamics-predicted state
532  // (aka process noise)
533  // Q* = E[(x_t - Ax_{t-1} - Bu_{t-1})*(x_t - Ax_{t-1} - Bu_{t-1})']
534  // t-1 terms:
535  View sum_e_x_tm1_x_tm1 =
536  sum_E_xu_tm1_xu_tm1_.submat(0, 0, n_x_ - 1, n_x_ - 1);
537  View sum_e_u_tm1_u_tm1 =
538  sum_E_xu_tm1_xu_tm1_.submat(n_x_, n_x_, n_x_ + n_u_ - 1, n_x_ + n_u_ - 1);
539  View sum_e_x_tm1_u_tm1 =
540  sum_E_xu_tm1_xu_tm1_.submat(0, n_x_, n_x_ - 1, n_x_ + n_u_ - 1);
541 
542  // t, t-1 terms:
543  View sum_e_x_t_x_tm1 = sum_E_xu_t_xu_tm1_.submat(0, 0, n_x_ - 1, n_x_ - 1);
544  View sum_e_x_t_u_tm1 =
545  sum_E_xu_t_xu_tm1_.submat(0, n_x_, n_x_ - 1, n_x_ + n_u_ - 1);
546 
547  Matrix q = sum_E_x_t_x_t_;
548  q += fit_.A() * sum_e_x_tm1_x_tm1 * fit_.A().t();
549  q -= sum_e_x_t_x_tm1 * fit_.A().t();
550  q -= fit_.A() * sum_e_x_t_x_tm1.t();
551  // input-related terms:
552  q += fit_.B() * sum_e_u_tm1_u_tm1 * fit_.B().t();
553  q -= sum_e_x_t_u_tm1 * fit_.B().t();
554  q -= fit_.B() * sum_e_x_t_u_tm1.t();
555  q += fit_.A() * sum_e_x_tm1_u_tm1 * fit_.B().t();
556  q += fit_.B() * sum_e_x_tm1_u_tm1.t() * fit_.A().t();
557  q /= n_t_tot_;
558 
559  fit_.set_Q(q);
560  std::cout << "Q_new[0]: " << fit_.Q()[0] << "\n";
561  // std::cout << "Q_new: \n" << fit_.Q() << "\n";
562 }

564 template <typename Fit>
565 void EM<Fit>::MaximizeInitial() {
566  Vector x0 = fit_.x0();
567  x0.zeros();
568  for (size_t trial = 0; trial < z_.size(); trial++) {
569  x0 += x_[trial].col(0);
570  }
571  x0 /= z_.size();
572  std::cout << "x0_new[0]: " << x0[0] << "\n";
573 
574  // always recalc P0 even if the initial state is fixed (at zero, for
575  // example)
576  Matrix e_var(n_x_, n_x_, fill::zeros);
577  for (size_t trial = 0; trial < z_.size(); trial++) {
578  e_var += (x_[trial].col(0) - x0) * (x_[trial].col(0) - x0).t();
579  }
580  e_var /= z_.size();
581 
582  // go ahead and subtract x0*x0' so don't have to below.
583  e_var -= x0 * x0.t();
584 
585  // To get P0, going to get initial P_ per trial and average.
586  // (which might be wrong, but need a single number)
587  Matrix p0 = fit_.P0();
588  p0.zeros();
589  for (size_t trial = 0; trial < z_.size(); trial++) {
590  p0 +=
591  (x_[trial].col(0) * x_[trial].col(0).t()) + P_[trial].slice(0) + e_var;
592  }
593  p0 /= z_.size();
594 
595  fit_.set_P0(p0);
596  std::cout << "P0_new[0]: " << fit_.P0()[0] << "\n";
597 }

599 template <typename Fit>
600 void EM<Fit>::MaximizeOutput() {
601  // solve for C+d:
602  Matrix sum_zx(n_y_, n_x_ + 1, fill::zeros);
603  Vector x1(n_x_ + 1, fill::zeros);
604  x1[n_x_] = 1.0; // augment with one to solve for bias
605  Matrix sum_e_x1_x1(n_x_ + 1, n_x_ + 1, fill::zeros);
606  for (size_t trial = 0; trial < z_.size(); trial++) {
607  for (size_t t = 1; t < n_t_[trial]; t++) {
608  x1.subvec(0, n_x_ - 1) = x_[trial].col(t);
609  sum_zx += z_.at(trial).col(t) * x1.t();
610  sum_e_x1_x1 += x1 * x1.t();
611  sum_e_x1_x1.submat(0, 0, n_x_ - 1, n_x_ - 1) += P_[trial].slice(t);

622 void EM<Fit>::MaximizeMeasurement() {
623  // Solve for measurement noise covar
624  size_t n_t_tot = 0;
625  // Ghahgramani, Hinton 1996:
626  Matrix sum_zz(n_y_, n_y_, fill::zeros);
627  Matrix sum_yz(n_y_, n_y_, fill::zeros);
628  for (size_t trial = 0; trial < z_.size(); trial++) {
629  for (size_t t = 1; t < n_t_[trial]; t++) {
630  sum_zz += z_.at(trial).col(t) * z_.at(trial).col(t).t();
631  // Use Cnew:
632  sum_yz +=
633  (fit_.C() * x_[trial].col(t) + fit_.d()) * z_.at(trial).col(t).t();
634  n_t_tot += 1;
635  }
636  }
637  fit_.set_R((sum_zz - sum_yz) / n_t_tot);
638  std::cout << "R_new[0]: " << fit_.R()[0] << "\n";
