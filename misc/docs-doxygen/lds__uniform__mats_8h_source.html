146 template <MatrixListFreeDim D>
147 inline void UniformMatrixList<D>::Swap(Matrix& that, size_t n) {
148  // make sure request in range
149  if (n >= this->size()) {
150  std::cerr
151  << "Requested UniformMatrixList element out of bounds. Skipping.\n";
152  return;
153  }
154  // check dim
155  bool does_match = true;
156  if (!(D == kMatFreeDim1)) {
157  does_match = does_match && (dim_[0][0] == that.n_rows);
158  }
159  if (!(D == kMatFreeDim2)) {
160  does_match = does_match && (dim_[0][1] == that.n_cols);
161  }
162  if (!does_match) {
163  std::cerr << "Cannot swap a UniformMatrixList element for an element of "
164  "different size. Skipping.\n";
165  return;
166  }
167  // if checks pass, perform swap
168  Matrix tmp = std::move((*this)[n]);
169  (*this)[n] = std::move(that);
170  that = std::move(tmp);
171 
172  if (D == kMatFreeDim1) {
173  this->dim_[n][0] = (*this)[n].n_rows;
174  }
175  if (D == kMatFreeDim2) {
176  this->dim_[n][1] = (*this)[n].n_cols;
177  }
178 }

183  // check dimensions
184  if (!this->empty()) {

186  std::ostringstream ss;

200  dims_nonzero = false;

206  if (!(D == kMatFreeDim1)) {

208  }

217  throw std::runtime_error(ss.str());

315 template <MatrixListFreeDim D>
316 void UniformMatrixList<D>::CheckDimensions(std::array<size_t, 2> dim) {
317  // change behavior based on free dim D
318  if ((dim[0] == 0) && !(D == kMatFreeDim1)) {
319  dim[0] = this->at(0).n_rows;
320  }
321  if ((dim[1] == 0) && !(D == kMatFreeDim2)) {
322  dim[1] = this->at(0).n_cols;
323  }
324 
325  // make sure dimensiolaties are all uniform
326  bool does_match(true);
327  for (const Matrix& mat : *this) {
328  if (!(D == kMatFreeDim1)) {
329  does_match = does_match && (mat.n_rows == dim[0]);
330  }
331  if (!(D == kMatFreeDim2)) {
332  does_match = does_match && (mat.n_cols == dim[1]);
333  }
334  if (!does_match) {
335  throw std::runtime_error(
336  "Dimensionality of one or more input matrices are not uniform.");
337  }
338  }
339 
340  dim_ = std::vector<std::array<size_t, 2>>(this->size(), dim);
341  for (size_t k = 0; k < this->size(); k++) {
342  dim_[k][0] = (*this)[k].n_rows;
343  dim_[k][1] = (*this)[k].n_cols;
